ASSIGNEMENT :9

INPUT STR1

INPUT STR2

FOR COUNT=1 TO LEN(STR1)
   IF LEN(STR1)>= 3 THEN
      S1=LEFT(STR1, 2)
   ELSE
      OUTPUT 'STRING 1 IS INVALID'
   ENDIF
NEXT

FOR COUNT = 1 TO LEN(STR2)
    IF LEN(STR2)>=3 THEN
       S2=LEFT(STR2,2)
    ELSE
       OUTPUT 'STRING 2 IS INVALID'
    ENDIF
    SFIRST = REPLACE(STR1,S1,S2)
    SLAST = REPLACE(STR2,S2,S1)
NEXT
OUTPUT SFIRST AND SLAST
END
